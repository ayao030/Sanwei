```python
def validate_input_data(front_coords, right_coords):
    """在处理前验证输入数据"""
    
    checks = []
    
    # 检查1：基本完整性
    if not front_coords or not right_coords:
        checks.append(("ERROR", "缺少视图数据"))
        return checks
    
    # 检查2：ID一致性
    front_ids = set(front_coords.keys())
    right_ids = set(right_coords.keys())
    
    missing_in_front = right_ids - front_ids
    missing_in_right = front_ids - right_ids
    
    if missing_in_front or missing_in_right:
        checks.append((
            "WARN",
            f"ID不一致：正面缺{len(missing_in_front)}条，"
            f"侧面缺{len(missing_in_right)}条"
        ))
    
    # 检查3：Z值一致性
    z_mismatches = []
    for gid in front_ids & right_ids:
        z_f1, z_f2 = front_coords[gid][0][1], front_coords[gid][1][1]
        z_r1, z_r2 = right_coords[gid][0][1], right_coords[gid][1][1]
        
        if abs(z_f1 - z_r1) > 0.5 or abs(z_f2 - z_r2) > 0.5:
            z_mismatches.append(gid)
    
    if z_mismatches:
        checks.append((
            "WARN",
            f"Z值不一致的杆件：{z_mismatches[:5]}..."
        ))
    
    # 检查4：极端值
    all_coords = []
    for seg in list(front_coords.values()) + list(right_coords.values()):
        all_coords.extend(seg)
    
    xs = [p[0] for p in all_coords]
    ys = [p[1] for p in all_coords]
    
    x_range = max(xs) - min(xs)
    y_range = max(ys) - min(ys)
    
    if x_range < 1 or y_range < 1:
        checks.append(("WARN", "坐标范围过小，可能是单位错误"))
    
    if x_range > 100 or y_range > 100:
        checks.append(("WARN", "坐标范围过大，请验证单位"))
    
    checks.append(("OK", f"验证通过：{len(front_ids)}条杆件"))
    return checks
```

```python
def comprehensive_validation(nodes, members):
    """完整的模型验证"""
    
    report = {
        "timestamp": datetime.now().isoformat(),
        "checks": {},
        "summary": {}
    }
    
    # 检查1：拓扑完整性
    missing_endpoints = 0
    for member in members:
        if member['node1_id'] not in nodes:
            missing_endpoints += 1
        if member['node2_id'] not in nodes:
            missing_endpoints += 1
    
    report["checks"]["missing_endpoints"] = {
        "status": "PASS" if missing_endpoints == 0 else "FAIL",
        "count": missing_endpoints
    }
    
    # 检查2：几何有效性
    zero_length_members = 0
    for member in members:
        n1 = nodes[member['node1_id']]
        n2 = nodes[member['node2_id']]
        dist = euclidean_distance(n1, n2)
        if dist < 1e-6:
            zero_length_members += 1
    
    report["checks"]["zero_length_members"] = {
        "status": "PASS" if zero_length_members == 0 else "FAIL",
        "count": zero_length_members
    }
    
    # 检查3：数值稳定性
    nan_count = 0
    inf_count = 0
    for node in nodes.values():
        for coord in ['X', 'Y', 'Z']:
            if math.isnan(node[coord]):
                nan_count += 1
            if math.isinf(node[coord]):
                inf_count += 1
    
    report["checks"]["numerical_stability"] = {
        "status": "PASS" if (nan_count == 0 and inf_count == 0) else "FAIL",
        "nans": nan_count,
        "infs": inf_count
    }
    
    # 检查4：坐标范围合理性
    xs = [n['X'] for n in nodes.values()]

    x_range = max(xs) - min(xs)

    }
    
    # 检查5：对称性验证（如果有对称杆件）
    symmetric_violations = 0
    # ... 对称性检查逻辑 ...
    
    # 生成总结
```

```python
# 优化前：低效的杆件遍历
def find_nearest_node_slow(target_point, nodes):
    min_dist = float('inf')
    nearest = None
    
    for node_id, node in nodes.items():
        dist = math.sqrt(
            (node['X'] - target_point[0])**2 +
            (node['Y'] - target_point[1])**2 +
            (node['Z'] - target_point[2])**2
        )
        if dist < min_dist:
            min_dist = dist
            nearest = node_id
    
    return nearest

# 优化后：使用KD树加速
from scipy.spatial import KDTree
import numpy as np

class OptimizedNodeSearch:
    def __init__(self, nodes):
        node_coords = np.array([
            [n['X'], n['Y'], n['Z']] for n in nodes.values()
        ])
        self.kdtree = KDTree(node_coords)
        self.node_list = list(nodes.keys())
    
    def find_nearest(self, target_point, k=1):
        _, indices = self.kdtree.query(target_point, k=k)
        if k == 1:
            return self.node_list[indices]
        else:
            return [self.node_list[i] for i in indices]

# 性能对比

nodes = {i: {'X': i*0.1, 'Y': i*0.05, 'Z': i*0.2} for i in range(1000)}

    lambda: find_nearest_node_slow(target, nodes),
    number=1000
)
print(f"原始方法：{time_slow*1000:.2f}ms")  # ~1500ms

# 优化方法
searcher = OptimizedNodeSearch(nodes)
time_fast = timeit.timeit(
    lambda: searcher.find_nearest(target),
    number=1000
)

print(f"加速比：{time_slow/time_fast:.1f}×")  # ~300×
```

```python
# 定义新的杆塔类型处理器
class CustomTowerProcessor:
    """处理特定类型的杆塔（如懸垂线塔）"""
    
    def __init__(self):
        self.special_rules = {}
    
    def register_member_classifier(self, member_id_pattern, classifier_func):
        """注册自定义的杆件分类规则"""
        self.special_rules[member_id_pattern] = classifier_func
    
    def classify_members(self, members):
        """应用自定义分类规则"""
        result = {
            'standard': [],
            'special': []
        }
        
        for member in members:
            classified = False
            
            # 尝试应用特殊规则
            for pattern, classifier in self.special_rules.items():
                if pattern.match(str(member['member_id'])):
                    result['special'].append({
                        'member': member,
                        'type': classifier(member)
                    })
                    classified = True
                    break
            
            # 如果没有特殊规则，使用标准分类
            if not classified:
                result['standard'].append(member)
        
        return result

# 使用示例
```
